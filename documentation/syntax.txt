statement 		:= reference_statement
           		| define_statement
           		| clean_statement
           		| print_statement 
           		| declare_statement
           		| (statement comment)*

print_stm 		:= 書之
clean_stm 		:= 噫
declare_stm 	:= 吾有NUM TYPE
reference_stm 	:= 名之曰 identifier
identifier 		:= 「(字)+」
comment 		:= 。

if_expr			:= 若condition者expr[若非expr]
pre_binary_op 	:= 加|減|乘|除
logic_math_op	:= 中有陽乎|中無陰乎
logic_if_op		:= 等於|不等於|不大於|不小於|大於|小於
post_math_op	:= 所餘幾何
pre_position	:= 以|於
unary_op 		:= 變
comment 		:= 注曰String
bool 			:= 陰|陽
